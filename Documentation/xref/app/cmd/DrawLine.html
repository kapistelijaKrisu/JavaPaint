8    *

12  public class DrawLine implements CMD {
13  
14      /**
15       *
16       * Draws a line based on area to img.
17       * @param img Target to be drawn on.
18       * @param area information on where to draw.
19       */
20      @Override
21      public void execute(MyImage img, TwoPoint area) {
22          img.getGraphics().draw(new Line2D.Float(area.getPrevX(), area.getPrevY(),
23                  area.getCurX(), area.getCurY()));
24      }
25  }
