5   import java.awt.Color;

8   public class SetAvgColor implements CMD {
9   
10      private PaintBrush brush;
11  
12      public SetAvgColor(PaintBrush brush) {
13          if (brush == null) {
14              throw new IllegalArgumentException();
15          }
16          this.brush = brush;
17      }
18  
19      @Override
20      public void execute(MyImage img, Area rect) {
21          if (rect == null || img == null) {
22              throw new IllegalArgumentException();
23          }
24          int a = 0;
25          int r = 0;
26          int g = 0;
27          int b = 0;
28  
29          for (int y = 0; y < rect.getCurY() + 1; y++) {
30              for (int x = 0; x < rect.getCurX() + 1; x++) {
31  
32                  int val = img.getImg().getRGB(x + rect.getStartX(), y + rect.getStartY());
33                  a += (0xff000000 & val) >>> 24;
34                  r += (0x00ff0000 & val) >> 16;
35                  g += (0x0000ff00 & val) >> 8;
36                  b += (0x000000ff & val);
37              }
38  
39          }
40  
41          int divider = (rect.getCurX() + 1) * (rect.getCurY() + 1);
42          a /= divider;
43          r /= divider;
44          g /= divider;
45          b /= divider;
46  
47          // System.out.println(a + " " + r + " " + g + " " + a);
48          brush.setColor(new Color(r, g, b, a));
49          brush.installSetting(img.getGraphics(), true, false, false);
50  
51      }
52  
53  }
