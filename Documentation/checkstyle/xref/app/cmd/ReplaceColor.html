7   /**
8    * 
9    * <p>An implementation of CMD which fills a replaces a color of MyImage to color it has been set.</p>
10   * 
11   */
12  public class ReplaceColor implements CMD {
13  
14      @Override
15      public void execute(MyImage img, Area area) {
16          int toReplace = img.getImg().getRGB(area.getCurX(), area.getCurY());
17          int height = img.getImg().getHeight();
18          int width = img.getImg().getWidth();
19          
20          for (int y = 0; y < height; y++) {
21              for (int x = 0; x < width; x++) {
22                  if (img.getImg().getRGB(x, y) == toReplace) {
23                      img.getImg().setRGB(x, y, img.getColor().getRGB());
24                  }
25                  
26              }
27              
28          }
29          
30      }
31      
32  }
