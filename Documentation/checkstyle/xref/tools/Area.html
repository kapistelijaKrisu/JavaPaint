3   /**
4    *
5    * <p>
6    * Class used to format data that implements CMD.</p>
7    * <p>
8    * Contains starting pixel, previous pixel, current pixel as int x,y</p>
9    *
10   */
11  public final class Area {
12  
13      private static final int MIN_VAL = 0;
14      private static int maxX = MIN_VAL, maxY = MIN_VAL;
15  
16      private int startX, startY;
17  
18      private int curX, curY;
19      private int lastX, lastY;
20  
21      public Area(int x, int y) {
22          init(x, y);
23      }
24  
25      /**
26       * 
27       * <p>private constructor made for setting parameters into x,y,width,height format by finding smallest x,y and taking the absolute difference of bigger x,y as width,height.</p>
28       * <p>
29       * @param x will be startX
30       * @param y will be startY
31       * @param width will be currentX and lastX
32       * @param height y will be currentY and lastY
33       */
34      private Area(int x, int y, int width, int height) {
35          startX = Math.min(x, width);
36          startY = Math.min(y, height);
37  
38          curX = fixToRange(Math.abs(x - width), true);
39          curY = fixToRange(Math.abs(y - height), false);
40  
41          lastX = curX;
42          lastY = curY;
43  
44      }
45  
46      /**
47       * 
48       * @param x - sets all x's
49       * @param y  sets all y's
50       * 
51       * checks values legality and fixes them with fixToRange()
52       */
53      public void init(int x, int y) {
54          x = fixToRange(x, true);
55          y = fixToRange(y, false);
56          lastX = x;
57          lastY = y;
58          startX = x;
59          startY = y;
60          curX = startX;
61          curY = startY;
62      }
63  
64      /**
65       * <p>parameters are checked and fixed with fixToRange()
66       * 
67       * @param x - new curX
68       * @param y - new curY</p>
69       * <p>
70       * prevX will be previous curX <br>
71       * prevY will be previous curX </p>
72       */
73      public void udpate(int x, int y) {
74          x = fixToRange(x, true);
75          y = fixToRange(y, false);
76          lastX = curX;
77          lastY = curY;
78          curX = x;
79          curY = y;
80      }
81  
82      /**
83       * 
84       * @param value value to be fixed.
85       * @param x - true = value is fixed according to x-axis, else value is fixed according to y-axis
86       * @return returns legal value of value. No changes will be made if value is legal. 
87       */
88      private int fixToRange(int value, boolean x) {
89          value = Math.max(MIN_VAL, value);
90          if (x) {
91              value = Math.min(value, maxX);
92          } else {
93              value = Math.min(value, maxY);
94          }
95          return value;
96      }
97  
98      public int getStartX() {
99          return startX;
100     }
101 
102     public int getStartY() {
103         return startY;
104     }
105 
106     public int getCurX() {
107         return curX;
108     }
109 
110     public int getCurY() {
111         return curY;
112     }
113 
114     public int getLastX() {
115         return lastX;
116     }
117 
118     public int getLastY() {
119         return lastY;
120     }
121 
122     /**
123      * 
124      * @param maxX maximum value that x's can have <br>
125      * @param maxY maximum value that y's can have
126      */
127     public static void setBounds(int maxX, int maxY) {
128         maxX = Math.max(MIN_VAL, maxX);
129         maxY = Math.max(maxY, MIN_VAL);
130         Area.maxX = maxX;
131         Area.maxY = maxY;
132     }
133 
134     /**
135      * 
136      * @return return itself in x,y,width, height format. See private constructor.
137      */
138     public Area getRectangle() {
139         return new Area(startX, startY, curX, curY);
140 
141     }
